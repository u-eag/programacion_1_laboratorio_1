int factorial (x)
{
int result;
int acumuladorFactorial=1;
int i;
    if(x>=0){
        for(i=x;i>1;i--){
            acumuladorFactorial=acumuladorFactorial*i;
        }
        result=acumuladorFactorial;
    } else {
        result=-1;
        // el factorial de un negativo no existe
    }
return result;
}
